<div id="get">
  ## جلب
</div>

جلب نسب الأرباح لقائمة من الرموز، أو أرقام التعريف الدولي للأوراق المالية ISIN، أو معرّفات CIK

<div id="example-usage">
  ### مثال على الاستخدام
</div>

```java
package hello.world;

import java.lang.Exception;
import org.benzinga.BZClient.Bzclient;
import org.benzinga.BZClient.models.operations.GetEarningRatiosV21Response;

public class Application {

    public static void main(String[] args) throws Exception {

        Bzclient sdk = Bzclient.builder()
                .apiKeyAuth("<YOUR_API_KEY_HERE>")
            .build();

        GetEarningRatiosV21Response res = sdk.earningRatios().get()
                .symbols("<value>")
                .asOf("<value>")
                .period("<value>")
                .reportType("<value>")
                .call();

        if (res.string().isPresent()) {
            // معالجة الرد
        }
    }
}
```
